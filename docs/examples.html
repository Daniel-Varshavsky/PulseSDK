Examples

End-to-end code samples and JSON request/response shapes.

Example 1 — an A/B tested button with star-rating feedback

class HomeActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_home)

        val ctaButton = findViewById<Button>(R.id.ctaButton)
        val variant = PulseSDK.getVariant("Homepage CTA")

        when (variant?.variantName) {
            "Light Theme" -> ctaButton.setBackgroundColor(Color.WHITE)
            "Dark Theme"  -> ctaButton.setBackgroundColor(Color.BLACK)
            else          -> { /* experiment not running — default styling */ }
        }

        ctaButton.setOnClickListener {
            showRatingDialog { rating ->
                variant?.let {
                    PulseSDK.submitStarRating(it.variantId, rating, screenId = "home")
                }
            }
        }
    }
}

Example 2 — general feedback form (not tied to an experiment)

submitFeedbackButton.setOnClickListener {
    val text = feedbackInput.text.toString()
    if (text.isNotBlank()) {
        PulseSDK.submitText(text, screenId = "feedback-screen")
        Toast.makeText(this, "Thanks for the feedback!", Toast.LENGTH_SHORT).show()
    }
}

Example 3 — multiple choice + metadata-driven UI

val variant = PulseSDK.getVariant("Navigation Style") ?: return
val choices = variant.choices ?: emptyList()   // e.g. ["Very clear", "Somewhat clear", ...]

choiceGroup.removeAllViews()
choices.forEachIndexed { index, label ->
    val radio = RadioButton(this).apply { text = label }
    radio.setOnClickListener {
        PulseSDK.submitMultipleChoice(variant.variantId, index)
    }
    choiceGroup.addView(radio)
}
